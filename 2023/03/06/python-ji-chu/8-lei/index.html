5. 继承 多继承

#继承  多继承
#方法  自己有调用自己的，自己无用上一级，选靠前的，无在选上一级，无报错
class base:
    def __init__(self,name):
        self.name = name
    def run(self):
        print('%s在跑步'%self.name)

class A(base):
    def play(self):
        print('%s在打球'%self.name)

class B(base):
    def chi(self):
        print('%s在吃饭'%self.name)

class C(A,B):
    def drink(self):
        print('%s在喝水'%self.name)
    def run(self):
        print('%s在第二次跑步' % self.name)

aa = A('wsd')
aa.run()
a = C('wsd2')
a.run()


#多继承
class AA:
    def arm(self):
        print('arm')

class BB:
    def leg(self):
        print('leg')

class CC:
    def been(self):
        print('been')

class person(AA,BB,CC):          #Mixin  开发模式
    def __init__(self,name):
        self.name = name
    def arm(self):
        print('arm2')
        super().arm()              #继承父类的数据，向上查找，直到找到为止

DD = person('wsd')
DD.arm()

print(person.__mro__)        #查找person的继承顺序
print(DD.__class__)         #通过实例查找

定义一个矩形类，有长河宽两个实例属性，还有一个计算面积的方法

# 定义一个矩形类，有长河宽两个实例属性，还有一个计算面积的方法
class rectangle:

    def __init__(self,length,width):
        self.length = length
        self.width = width

    def count(self):
        area = self.length * self.width
        return area

re = rectangle(3,4)
area = re.count()
print(area)
