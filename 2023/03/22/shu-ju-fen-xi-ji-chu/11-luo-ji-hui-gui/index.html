import numpy as np
import matplotlib.pyplot as plt

plt.rcParams['font.sans-serif'] = [u'simHei']
plt.rcParams['axes.unicode_minus'] = False
x = np.linspace(-10, 10) # 默认取50个数值
y = 1 / (1 + np.exp(-x)) # sigmoid函数
y_ = y * (1 - y) # sigmoid的导数
plt.plot(x, y, 'r-', label='sigmoid函数')
plt.plot(x, y_, 'r--', label='sigmoid导数')
plt.scatter(0,0.5)
plt.legend()
plt.grid()
plt.show()

3. 逻辑回归案例

import numpy as np
from sklearn.linear_model import LogisticRegression
import matplotlib.pyplot as plt
import warnings
warnings.filterwarnings('ignore')#防止anaconda报警告性错误

# data数据   4,3    x1, x2, y
data = np.array([
    [1, 1, 0],
    [1, 2, 0],
    [0, 0, 1],
    [-1, 0, 1]
])
#数据集切分，  前两列特征，最后一列作为标签
x = data[:, :-1]
y = data[:, -1:]
print(x)
print(y)
[[ 1  1]
 [ 1  2]
 [ 0  0]
 [-1  0]]
[[0]
 [0]
 [1]
 [1]]

model = LogisticRegression()
model.fit(x, y.ravel())
y_ = model.predict(x)
print(y_)  # [0 0 1 1]

# 逻辑回归将数值缩放到0-1范围之内，代表事件发生的可能性
# 打印输出的可能性， 第一列为0样本，第二列为1样本
print(model.predict_proba(x))
[[0.73559819 0.26440181]
 [0.83458047 0.16541953]
 [0.43165661 0.56834339]
 [0.27325413 0.72674587]]

w = model.coef_
b = model.intercept_
print(w)
print(b)
[[-0.70307921 -0.59523025]]
[ 0.27509539]

min_x=min(x[:,0])   #x1的最小值
max_x=max(x[:,1])   #x1的最大值
y_min_x=(-b-w[:,0]*min_x)/w[:,1]     #x1的最小值对应的x2值
y_max_x=(-b-w[:,0]*max_x)/w[:,1]     #x1的最大值对应的x2值
plt.plot([min_x, max_x], [y_min_x, y_max_x],'-g')   #画分界线
for i in range(len(data)):
    if y_[i] == 0:
        plt.scatter(x[i,0], x[i,1], marker='*', c='r')
    else:
        plt.scatter(x[i,0], x[i,1], marker='^', c='y')
plt.grid()
plt.show()

结论，样本点距离线性模型的区分线越近，就证明概率越不确定

7. 乳腺癌分类案例

import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix, classification_report, roc_auc_score, roc_curve  # 混搅矩阵 分类报告 rocauc得分 roc曲线

# 数据加载处理
#数据的路径
path = "../datas/breast-cancer-wisconsin.data"
#给数据中每一列数据进行起名字
names = ['id','Clump Thickness','Uniformity of Cell Size','Uniformity of Cell Shape',
         'Marginal Adhesion','Single Epithelial Cell Size','Bare Nuclei',
        'Bland Chromatin','Normal Nucleoli','Mitoses','Class']
# 没有字段名，所以header设为None， 并且给每一列数据起名字
df = pd.read_csv(path,sep=",",header=None,names=names)
# 将标签结果由4,2，转化为1,0
df[['Class']] = df[['Class']]/2 - 1
# 清除有缺失值数据
df = df.replace("?",np.nan)
df = df.dropna(how="any")
# 统计类别频数
print(df['Class'].value_counts())
0.0    444
1.0    239
Name: Class, dtype: int64
        
# 可以看到现在数据的标签数量差距较大，使用过采样处理，将数据类别调整平衡
df1 = df[df.Class==1]
#DataFrame.sample   n 要抽取的行数  frac 抽取比例   replace是否又放回采样
df1 = df1.sample(frac=(444-239)/239)
df = pd.concat([df,df1],axis=0)
