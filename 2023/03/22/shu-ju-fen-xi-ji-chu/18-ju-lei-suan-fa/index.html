1. 聚类算法-K-means

1.1 K-means底层算法实现

import numpy as np
import matplotlib.pyplot as plt
import copy
plt.rcParams['font.sans-serif'] = [u'SimHei']
plt.rcParams['axes.unicode_minus'] = False

# 处理前效果演示
# 质心位置为随机样本点
# 样本集
X = np.array([[1, 2], [2, 2], [6, 8],[7 ,8]]) #shape(4, 2)
#定义初始化质心
C = np.array([[1.0, 2.0], [2.0, 2.0]])
plt.scatter(X[:, 0], X[:, 1], label='样本点') #X[:, 0]处理每个样本点第一个特征
plt.scatter(C[:, 0], C[:, 1], c="red", marker='*', label='质心')
plt.title('处理前效果')
plt.legend()
plt.show()

底层实现核心内容

1.计算所有样本点到每个质心距离
2.比较样本点距离哪一个质心更近，就记为哪一个簇
3.使用当前簇内的样本，采用均值方式，更新质心位置

iters = 5 #循环次数，如果说质心没有确定下来，模型只执行5次
x = 0 # 统计迭代的次数
while (x < 5) :
    iters -= 1
    x += 1
    B = [] #每个样本点到质心的距离
    C_ = copy.deepcopy(C) # 深层拷贝，质心需要更新，停止条件质心不再变化， 后续要比较C_ == C是否相等
    for c in C:
        #计算每个点到质心的欧式距离
        dis = np.sqrt(((X - c)**2).sum(axis=1))
        # print(dis)
        B.append(dis)
    #求样本点属于哪一个类别
    # print(B)
    min_idx = np.argmin(np.array(B),axis=0)
    # print(min_idx)
    for i in range(len(C)):
        #更换每个质心的位置
        C[i] = np.mean(X[min_idx == i],axis=0)
    if np.sum(C - C_) == 0:# 终止条件，使用之前的质心位置和当前质心位置进行比较，如果没有变化，就停止迭代
        break
    plt.scatter(X[:, 0], X[:, 1], c=min_idx, label='样本点')
    plt.scatter(C[:, 0], C[:, 1], c="red", marker='*', label='质心')
    plt.title('第%d次处理后的聚类效果'%x)
    plt.legend()
    plt.show()

print(min_idx)
print(C)

2. K-means衍生算法

2.4 性能评估

2.5 K-means与K-means++对比-代码

import numpy as np
import matplotlib.pyplot as plt
import sklearn.datasets as ds     ## 随机数据生成
import matplotlib.colors
from sklearn.cluster import KMeans

# 指定输出的边界
def expand(a, b):
    return a-1, b+1#最小值-1，最大值+1

N = 400#400个样本点
centers = 4 # 4个中心
# 样本点数量，特征数，中心数，随机种子
data, y = ds.make_blobs(N, n_features=2, centers=centers, random_state=6)

matplotlib.rcParams['font.sans-serif'] = [u'SimHei']
matplotlib.rcParams['axes.unicode_minus'] = False
cm = matplotlib.colors.ListedColormap(list('rgbm'))# 指定四个中心的颜色，红绿蓝，粉

plt.title(u'原始数据')
plt.scatter(data[:, 0], data[:, 1], c=y, s=30, cmap=cm)
x1_min, x2_min = np.min(data, axis=0)
x1_max, x2_max = np.max(data, axis=0)
x1_min, x1_max = expand(x1_min, x1_max)
x2_min, x2_max = expand(x2_min, x2_max)
plt.xlim((x1_min, x1_max))
plt.ylim((x2_min, x2_max))
plt.grid()
plt.show()

# Kmeans的调用， 默认形式是采用K-means++进行处理的，如果使用K-means操作，就需要init='random'
cls = KMeans(n_clusters=4, init='random')
y_hat = cls.fit_predict(data)
plt.title(u'KMeans聚类')
plt.scatter(data[:, 0], data[:, 1], c=y_hat, s=30, cmap=cm)
plt.xlim((x1_min, x1_max))
plt.ylim((x2_min, x2_max))
plt.grid()
plt.show()

# K-means++调用
cls = KMeans(n_clusters=4, init='k-means++')
y_hat = cls.fit_predict(data)
plt.title(u'KMeans++聚类')
plt.scatter(data[:, 0], data[:, 1], c=y_hat, s=30, cmap=cm)
plt.xlim((x1_min, x1_max))
plt.ylim((x2_min, x2_max))
plt.grid()
plt.show()

# 画图查找最优K点
# 当k小于真实聚类数时，由于k的增大会大幅增加每个簇的聚合程度，故SSE的下降幅度会很大，而当k到达真实聚类数时，再增加k所得到的聚合程度回报会迅速变小，所以SSE的下降幅度会骤减，然后随着k值的继续增大而趋于平缓，也就是说SSE和k的关系图是一个手肘的形状! 可以看出k=3的效果是最好效果
record1 = [] #记录每一次生成SSE
record2 = [] #轮廓系数
for i in range(2,10): # k值范围2-9
    kmeans = KMeans(i)
    kmeans.fit(data)
    record1.append(kmeans.inertia_) #inertia簇内误差平方和  SSE    下降最大 中 小 中点
    label = kmeans.labels_
    record2.append(silhouette_score(data,label)) #轮廓系数    轮廓系数接近与1的点
plt.plot(range(2,10),record1, range(2,10),record1, marker='o', c='b')
plt.show()# k=3是最优k值点
plt.plot(range(2,10),record2, range(2,10),record2, marker='o', c='b')
plt.show()# k=3是最优k值点

4. 密度聚类-DBSCAN

密度聚类案例

基于密度进行计算聚类
可以处理非凸数据
可以查找噪声点

import numpy as np
import matplotlib.pyplot as plt
from sklearn import datasets     # 生成数据
from sklearn.cluster import KMeans
from sklearn.cluster import DBSCAN

# 使用随机样本点生成初始样本状态
plt.rcParams['font.sans-serif']=['SimHei']
plt.rcParams['axes.unicode_minus']=False
X1, y1=datasets.make_circles(n_samples=5000, factor=.6, noise=.05)
X2, y2 = datasets.make_blobs(n_samples=1000, n_features=2, centers=[[1.2,1.2]], 
                             cluster_std=[[.1]], random_state=9)
X = np.concatenate((X1, X2))
plt.scatter(X[:, 0], X[:, 1], marker='o')
plt.title('原始数据样本点分布')
plt.show()
print(X1[0:10,:])
print(X2[0:10,:])
print(X[0:10,:])

# 使用K-means进行样本聚类，啧啧，效果不太好
y_pred = KMeans(n_clusters=3, random_state=9).fit_predict(X)
plt.scatter(X[:, 0], X[:, 1], c=y_pred)
plt.title('K-means算法进行聚类处理')
plt.show()
print(y_pred[0:20])

# 使用密度聚类DBSCAN，会发现，如果不设置半径和数量很难有效生成簇
y_pred = DBSCAN().fit_predict(X)
plt.scatter(X[:, 0], X[:, 1], c=y_pred)
plt.title('默认参数密度聚类')
plt.show()

# 如果通过调整半径和样本数量，可以有效体现聚类效果
y_pred = DBSCAN(eps = 0.08, min_samples=3).fit_predict(X)#eps = 0.08 半径, min_samples=3样本数量
plt.scatter(X[:, 0], X[:, 1], c=y_pred)
plt.title('设置参数的密度聚类')
plt.show()

5. 层次聚类

层次聚类 凝聚聚类

import pandas as pd
from sklearn.cluster import AgglomerativeClustering
import matplotlib.pyplot as plt
import scipy.cluster.hierarchy as sch

dict = {'x1':[16.9,38.5,39.5,80.8,82,34.6,116.1]}
data = pd.DataFrame(dict)
print(data)
	x1
0   16.9
1   38.5
2   39.5
3   80.8
4   82.0
5   34.6
6  116.1

# 将聚类分成三个类别，并对聚类结果进行聚类处理
ac = AgglomerativeClustering(n_clusters=3) #凝聚聚类
ac.fit(data)
data['cluster'] = ac.labels_ #将聚类结果进行处理，分别对应样本
print(data)
x1  cluster
0   16.9        0
1   38.5        0
2   39.5        0
3   80.8        1
4   82.0        1
5   34.6        0
6  116.1        2

# 生成可视化效果图
# 生成点与点之间的距离矩阵,这里用的欧氏距离:
dis = sch.distance.pdist(data, 'euclidean')
# 将层级聚类结果以树状图表示出来
model = sch.linkage(dis, method='average')
pic = sch.dendrogram(model,labels=['A','B','C','D','E','F','G'])
plt.show()
