# 写一个函数实现此功能
# 将score进行处理，大于60 修改为及格   否则为不及格

def fn(x):
    '''
    大于60 修改为及格   否则为不及格
    args:
        x:df中输入的信息
    return:
        及格 不及格 pass fail
    '''
    if x>= 60:
        return 'pass'
    else:
        return 'fail'
    
df1['score'] = df1.score.map(fn)
