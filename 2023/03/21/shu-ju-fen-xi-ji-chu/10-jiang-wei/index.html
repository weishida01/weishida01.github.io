缺失值填充

缺失值是计算模型时最大的绊脚石，如果出现缺失值的情况，模型就不能正常进行数据拟合，会报错导致程序无法运行，所以我们需要对缺失值进行填充

import numpy as np
from sklearn.preprocessing import Imputer
import warnings
warnings.filterwarnings('ignore')#防止anaconda报警告性错误

x = np.array([
    [2, 2, 4, 1], 
    [np.nan, 3, 4, 4],
    [1, 1, 1, np.nan],
    [2, 2, np.nan, 3]
])
print(x)
[[ 2.  2.  4.  1.]
 [nan  3.  4.  4.]
 [ 1.  1.  1. nan]
 [ 2.  2. nan  3.]]

拟合的数据采用x的数据进行操作，这里是按照列操作
所有的预测值只根据非空数值计算，做均值处理就是求出所有非空数值的和，再除以非空数值的数量，得到均值

imp1 = Imputer(missing_values='NaN', strategy='mean', axis=0)#处理缺失值nan填充策略 均值  回归类型，axis=0横向比较
imp1.fit(x)#拟合
print(imp1.transform(x))#转化

二值化处理

为了保证能够更明确的表示特征的处理后的结果，一般情况下可以把连续性问题调整为离散型特征
只要通过设定阈值的方式就可以进行调整，大于阈值返回1，小于阈值返回0

from sklearn.preprocessing import Binarizer

a = np.array([
    [1, 1.5, 2.3],
    [0.6, 3, 2.0]
])
model = Binarizer(threshold=2)#  阈值：考试通过是2分，如果低于2，就为0， 高于2就为1
print(model.fit_transform(a))
[[0. 0. 1.]
 [0. 1. 0.]]

7. SVD应用

from PIL import Image
import numpy as np
import matplotlib as mpl
import matplotlib.pyplot as plt

# 中文显示问题
mpl.rcParams['font.sans-serif'] = [u'simHei']
mpl.rcParams['axes.unicode_minus'] = False

A = Image.open(r'../datas/godness.jpg')
a = np.array(A) 
print(a.shape)  # (432, 600, 3)

# 由于是彩色图像，所以3通道。a的最内层数组为三个数，分别表示RGB，用来表示一个像素
u_r, sigma_r, v_r = np.linalg.svd(a[:, :, 0])#红色通道图片
u_g, sigma_g, v_g = np.linalg.svd(a[:, :, 1])#绿色通道图片
u_b, sigma_b, v_b = np.linalg.svd(a[:, :, 2])#蓝色通道图片
# 然后便可以根据需要压缩图像（丢弃分解出来的三个矩阵中的数据）， 利用的奇异值个数越少，则压缩的越厉害。下面来看一下不同程度压缩后，重构图像的清晰度：

def restore1(u, sigma, v, k):  # u 左奇异矩阵，  sigma： 特征值矩阵，   v  右奇异矩阵   k：选择的奇异值数量   10% 就会有很好的图像显示效果
    # 重构图像
    a = np.dot(u[:, :k], np.diag(sigma[:k])).dot(v[:k, :])
    #图像的像素点数值显示范围是在0-255之间， 
    a[a < 0] = 0
    a[a > 255] = 255
    return np.rint(a).astype('uint8')#为了保证处理后的结果是一个整数

plt.figure(facecolor='w', figsize=(10, 10))#图片尺寸，底片白色
# 奇异值个数依次取：1,2,...,20看看一下效果
K = 20
for k in range(1, K + 1):#1-20
    R = restore1(u_r, sigma_r, v_r, k)
    G = restore1(u_g, sigma_g, v_g, k)
    B = restore1(u_b, sigma_b, v_b, k)
    I = np.stack((R, G, B), axis=2)#通道是第三个维度所以axis=2
    # 将图片重构后的显示出来
    plt.subplot(5, 4, k)
    plt.imshow(I)
    plt.axis('off')
    plt.title(u'奇异值个数：%d' % k)

plt.suptitle(u'SVD与图像分解', fontsize=20)
plt.tight_layout(0.1, rect=(0, 0, 1, 0.92))
plt.show()

9. PCA代码

import numpy as np

def pca(X,k):
    m_samples , n_features = X.shape
    #中心化  去均值    均值为0
    mean = np.mean(X,axis=0)
    normX = X - mean #去均值，中心化
    cov_mat = np.dot(np.transpose(normX),normX)  #协方差矩阵
    #对二维数组的transpose操作就是对原数组的转置操作  矩阵相乘
    vals , vecs = np.linalg.eig(cov_mat)# 得到特征向量和特征值
    print('特征值', vals)
    print('特征向量', vecs)
    eig_pairs = [(np.abs(vals[i]),vecs[:,i]) for i in range(n_features)]
    print(eig_pairs)
    print('----------')
    #将特征值由大到小排列
    eig_pairs.sort(reverse=True)
    print(eig_pairs)
    print('------------')
    #获取多少个维度
    feature = np.array(eig_pairs[0][k])
    print(feature)
    #将数据进行还原操作  normX  中心化后的数据 和  特征向量相乘
    data = np.dot(normX,np.transpose(feature))
    return data

X = np.array([[-1,1],[-2,-1],[-3,-2],[1,1],[2,1],[3,2]])

10. 代码实战

SVD和PCA都是常用的数据降维处理方式
一般可以处理多特征数据的降维处理操作
SVD和PCA的实战开发使用量不相上下，没有优劣可研
由于本案例特征较少无法体现降维优势，只是做降维操作演示

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.decomposition import PCA, TruncatedSVD
from sklearn.metrics import mean_squared_error

data = pd.read_csv(r'../datas/Advertising.csv')    # TV、Radio、Newspaper、Sales
x = data[['TV', 'Radio', 'Newspaper']]
y = data['Sales']

# # 绘制样本分布，观察效果
plt.plot(data['TV'], y, 'ro', label='TV')
plt.plot(data['Radio'], y, 'g*', label='Radio')
plt.plot(data['Newspaper'], y, 'bv', label='Newspaer')
plt.legend(loc='lower right')
plt.grid()
plt.show()

##使用PCA降维后观察效果
pca = PCA(n_components=1)#我只选择一个维度， 下面显示的图像，和上面的图像不是吻合
x_ = pca.fit_transform(x)
print(x_.shape)
plt.plot(x_, y, 'g*')
plt.grid()
plt.show()

model = LinearRegression(fit_intercept=True)
model.fit(x_, y)
y_ = model.predict(x_)
print(mean_squared_error(y_pred=y_, y_true=y))  # 10.4431524615

svd = TruncatedSVD(n_components=1)
x_ = svd.fit_transform(x)
print(x_.shape)   # (200, 1)
plt.plot(x_, y, 'g*')
plt.grid()
plt.show()

model = LinearRegression(fit_intercept=True)
model.fit(x_, y)
y_ = model.predict(x_)
print(mean_squared_error(y_pred=y_, y_true=y)) # 9.7480425958
